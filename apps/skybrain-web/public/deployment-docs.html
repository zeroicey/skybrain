1. 系统架构概述

Skybrain 智能无人机巡检系统采用微服务架构设计，由前端、后端、模型训练、模型推理、视频推流五大核心模块组成。各服务通过 Docker 容器化部署，支持横向扩展。

from ultralytics import YOLO
import torch
import os

def main():
    # 检查 GPU 可用性
    print(f"CUDA available: {torch.cuda.is_available()}")
    if torch.cuda.is_available():
        print(f"GPU: {torch.cuda.get_device_name(0)}")

    # 加载预训练模型
    model = YOLO('yolov8n.pt')

    # 训练参数配置
    results = model.train(
        data='dataset.yaml',
        epochs=100,
        imgsz=640,
        batch=16,
        device=0,  # GPU 0
        project='/workspace/models',
        name='campus_detection',
        exist_ok=True,
        optimizer='SGD',
        lr0=0.01,
        lrf=0.01,
        momentum=0.937,
        weight_decay=0.0005,
        warmup_epochs=3,
        augment=True,
        mosaic=1.0,
        mixup=0.15,
        copy_paste=0.1,
        verbose=True,
    )

    # 保存最佳模型
    best_model_path = f"{results.save_dir}/weights/best.pt"
    print(f"Training complete. Best model: {best_model_path}")

if __name__ == '__main__':
    main()

5.3.2 推理 API 服务

Python model-inference/main.py 
from fastapi import FastAPI, UploadFile, File, HTTPException
from fastapi.responses import JSONResponse
from ultralytics import YOLO
import torch
import cv2
import numpy as np
from typing import List, Dict
import os

app = FastAPI(title="YOLO Inference API", version="1.0.0")

# 加载模型
MODEL_PATH = os.getenv("MODEL_PATH", "/models/best.pt")
model = None

@app.on_event("startup")
def load_model():
    global model
    print(f"Loading model from {MODEL_PATH}")
    print(f"CUDA available: {torch.cuda.is_available()}")
    model = YOLO(MODEL_PATH)
    if torch.cuda.is_available():
        model.to('cuda')
    print("Model loaded successfully")

@app.get("/health")
def health_check():
    return {"status": "healthy", "cuda_available": torch.cuda.is_available()}

@app.post("/detect")
async def detect_objects(file: UploadFile = File(...)):
    """检测上传图片中的目标"""
    if model is None:
        raise HTTPException(status_code=503, detail="Model not loaded")

    # 读取图片
    contents = await file.read()
    nparr = np.frombuffer(contents, np.uint8)
    image = cv2.imdecode(nparr, cv2.IMREAD_COLOR)

    # 执行检测
    results = model(image, conf=0.25, iou=0.45)

    # 解析结果
    detections = []
    for result in results:
        boxes = result.boxes
        for box in boxes:
            detections.append({
                "class": result.names[int(box.cls)],
                "confidence": float(box.conf),
                "bbox": {
                    "x1": float(box.xyxy[0][0]),
                    "y1": float(box.xyxy[0][1]),
                    "x2": float(box.xyxy[0][2]),
                    "y2": float(box.xyxy[0][3])
                }
            })

    return JSONResponse({
        "image_size": {"width": image.shape[1], "height": image.shape[0]},
        "detections": detections,
        "count": len(detections)
    })

@app.post("/detect/batch")
async def detect_batch(files: List[UploadFile] = File(...)):
    """批量检测多张图片"""
    if model is None:
        raise HTTPException(status_code=503, detail="Model not loaded")

    results = []
    for file in files:
        contents = await file.read()
        nparr = np.frombuffer(contents, np.uint8)
        image = cv2.imdecode(nparr, cv2.IMREAD_COLOR)

        detections = model(image, conf=0.25)
        results.append({
            "filename": file.filename,
            "detections": detections[0].boxes.shape[0]
        })

    return {"results": results}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)

6.2 WebSocket 连接管理

TypeScript apps/videopusher/src/index.ts 
import { WebSocketServer, WebSocket } from 'ws';
import { createServer } from 'http';

const PORT = parseInt(process.env.PORT || '9000');

interface Client {
    id: string;
    ws: WebSocket;
    droneId: string;
}

const wss = new WebSocketServer({ port: PORT });
const clients = new Map();

wss.on('connection', (ws, req) => {
    const clientId = generateId();
    console.log(`Client connected: ${clientId}`);

    ws.on('message', async (data) => {
        const message = JSON.parse(data.toString());

        switch (message.type) {
            case 'register':
                // 客户端注册
                clients.set(clientId, {
                    id: clientId,
                    ws,
                    droneId: message.droneId
                });
                break;

            case 'video_frame':
                // 转发视频帧给订阅者
                broadcastToSubscribers(message.droneId, {
                    type: 'video_frame',
                    frame: message.frame,
                    timestamp: message.timestamp
                });
                break;

            case 'detection_result':
                // 转发检测结果
                broadcastToSubscribers(message.droneId, {
                    type: 'detection_result',
                    detections: message.detections,
                    timestamp: message.timestamp
                });
                break;
        }
    });

    ws.on('close', () => {
        clients.delete(clientId);
        console.log(`Client disconnected: ${clientId}`);
    });
});

function broadcastToSubscribers(droneId: string, message: object) {
    clients.forEach((client) => {
        if (client.droneId === droneId && client.ws.readyState === WebSocket.OPEN) {
            client.ws.send(JSON.stringify(message));
        }
    });
}

console.log(`WebSocket server running on port ${PORT}`);

8. Docker Compose 配置

使用 Docker Compose 编排所有服务，一键启动完整系统。

8.1 docker-compose.yml

YAML docker-compose.yml 
version: '3.8'

services:
  # PostgreSQL 数据库
  postgres:
    image: postgres:15-alpine
    container_name: skybrain-postgres
    restart: unless-stopped
    environment:
      POSTGRES_USER: skybrain
      POSTGRES_PASSWORD: ${DB_PASSWORD:-change-me-in-production}
      POSTGRES_DB: skybrain
    volumes:
      - postgres_data:/var/lib/postgresql/data
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U skybrain"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: skybrain-redis
    restart: unless-stopped
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    ports:
      - "6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # 后端 API 服务
  backend:
    build:
      context: ./apps/api
      dockerfile: Dockerfile
    container_name: skybrain-backend
    restart: unless-stopped
    ports:
      - "3000:3000"
    env_file:
      - ./apps/api/.env
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    volumes:
      - ./apps/api/src:/app/src

  # 前端 Web 服务
  frontend:
    build:
      context: ./apps/skybrain-web
      dockerfile: ../frontend/Dockerfile
    container_name: skybrain-frontend
    restart: unless-stopped
    ports:
      - "80:80"
    depends_on:
      - backend
    environment:
      - VITE_API_BASE_URL=http://localhost/api
      - VITE_WS_URL=ws://localhost/ws

  # 视频推流服务
  videopusher:
    build:
      context: ./apps/videopusher
      dockerfile: Dockerfile
    container_name: skybrain-videopusher
    restart: unless-stopped
    ports:
      - "9000:9000"
    volumes:
      - ./apps/videopusher/config.yaml:/app/config.yaml:ro

  # YOLO 推理服务（GPU 加速）
  model-inference:
    build:
      context: ./model-inference
      dockerfile: Dockerfile
    container_name: skybrain-model-inference
    restart: unless-stopped
    ports:
      - "8000:8000"
    environment:
      - MODEL_PATH=/models/best.pt
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: 1
              capabilities: [gpu]
    volumes:
      - model_storage:/models
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
      interval: 30s
      timeout: 10s
      retries: 3

volumes:
  postgres_data:
  redis_data:
  model_storage:

networks:
  default:
    name: skybrain-network

推理服务配置了 NVIDIA GPU 支持。请确保主机已安装 NVIDIA Driver 和 NVIDIA Docker 运行时。

10. 常见问题排查

10.1 数据库连接失败

症状

后端日志显示 Connection refused 或 Authentication failed

  • 检查 PostgreSQL 容器状态：docker compose ps postgres
  • 验证数据库凭据：确认 .env 中的用户名密码正确
  • 等待数据库完全启动：首次启动可能需要 10-30 秒初始化

10.2 GPU 不可用

症状

推理服务日志显示 CUDA available: False

  • 验证 NVIDIA Driver：nvidia-smi
  • 检查 NVIDIA Docker 安装：docker run --rm --gpus all nvidia/cuda:12.1.0-base-ubuntu22.04 nvidia-smi
  • 确认 docker-compose.yml 中配置了 GPU 设备

10.3 前端无法连接后端

症状

浏览器控制台显示 CORS 或 Network Error

  • 检查 Nginx 代理配置：确认 /api/ 正确指向后端
  • 验证后端运行状态：curl http://localhost:3000/health
  • 检查后端 CORS 配置：确认允许前端域名的跨域请求

10.4 WebSocket 连接失败

症状

前端视频无法播放，显示连接失败

  • 检查 WebSocket 服务状态：docker compose ps videopusher
  • 验证 Nginx WebSocket 代理配置：需要 Upgrade 和 Connection 头
  • 检查防火墙：确认 9000 端口已开放

10.5 模型加载失败

症状

推理服务无法启动，提示模型文件不存在

  • 确认模型文件存在：docker exec skybrain-model-inference ls -la /models/
  • 检查模型文件权限：确保 /models/ 目录可读
  • 验证模型格式：YOLO 模型应为 .pt 格式

10.6 性能问题

性能优化建议
  • GPU 加速：推理服务必须使用 GPU 以满足实时性要求
  • 内存优化：增加 Docker 内存限制避免 OOM
  • 网络优化：使用内网通信避免网络延迟
  • 数据库优化：为常用查询添加索引
  • 缓存策略：合理使用 Redis 缓存减少数据库压力
